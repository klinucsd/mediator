 24# This WFS loader is designed with two key objectives:
 25#
 26# Resource Efficiency: To mitigate the risk of excessive memory usage and prevent potential system crashes when
 27# dealing with large WFS datasets. The loader ensures that only manageable portions of data are read at a time,
 28# preventing resource exhaustion.
 29#
 30# Concurrent Loading: To enhance overall loading speed, the loader employs a concurrent approach. It reads and
 31# processes smaller chunks of WFS data simultaneously, allowing for faster and more efficient loading of the entire
 32# dataset. This concurrent strategy optimizes resource utilization and reduces the overall loading time.

 34def save_gml_to_db(gml_binary, table_name, mode):
 35    with tempfile.NamedTemporaryFile(suffix=".gml", mode="wb") as temp:
 36        temp.write(gml_binary)
 37        temp.flush()  # Ensure data is written to the file
 38
 39        command = [
 40            "ogr2ogr",
 41            f"-{mode}",
 42            "-f", "PostgreSQL",
 43            f"PG:dbname={config('db_name')} host={config('db_host')} port={config('db_port')} user={config('db_user')} password=password",
 44            temp.name,
 45            "-nln", table_name
 46        ]
 47        try:
 48            result = subprocess.run(command, check=True, capture_output=True, text=True)
 49            print(result.stdout)  # Print successful output
 50        except subprocess.CalledProcessError as error:
 51            print("Error:", error)
 52            print("Command output:", error.output)  # Access error out
 53            raise DataLoaderError(f'Error when saving GML to PostGIS: {error}')

 56# This function is used by a new spawned process to save WFS_LOAD_FEATURES_PER_PROCESS
 57# features starting from start_index to PostGIS
 58def process_load_features(self_url, base_url, version, type_name, epsg_code, start_index,
 59                          sort_by, table_name, output_format, vendor, error_event):
 60    """
 61        Load features from a Web Feature Service (WFS) into a PostgresSQL/PostGIS database.
 62
 63        Args:
 64            self_url (str): The original URL for error logging
 65            base_url (str): The base URL of the WFS service.
 66            version (str): The version of the WFS service (e.g., '1.1.0', '2.0.0').
 67            type_name (str): The name of the feature type to retrieve from the WFS.
 68            epsg_code (str): The EPSG code representing the coordinate reference system (CRS).
 69            start_index (int): The starting index for retrieving features in a paginated manner.
 70            sort_by (str): The attribute to use for sorting features.
 71            table_name (str): The name of the PostgresSQL table to store the features.
 72            output_format (str): The name for the JSON output format.
 73            vendor (str): The name of the server vendor.
 74            error_event (threading.Event): An event to signal if an error occurs during the data loading process.
 75
 76        Raises:
 77            DataLoaderError: If the maximum number of retries is reached and the data loading process fails.
 78
 79        Note:
 80            This function attempts to load features from the specified WFS service into a PostgreSQL/PostGIS database.
 81            It uses pagination to load features in chunks and retries in case of errors.
 82
 83        Example:
 84            process_load_features(
 85                url='https://example.com/wfs',
 86                version='2.0.0',
 87                type_name='roads',
 88                epsg_code='4326',
 89                start_index=0,
 90                sort_by='objectid',
 91                table_name='roads_table',
 92                output_format='GEOJSON',
 93                vendor='ArcGIS',
 94                error_event=my_error_event
 95            )
 96    """
 97    # Set the number of retries in case of an error during loading
 98    tries = 0
 99
100    # Retry loading features in case of an error or no error
101    while tries < DATA_LOAD_RETRIES_ON_ERROR:
102        try:
103            # Create a WebFeatureService instance
104            wfs = WebFeatureService(base_url, version, timeout=120)
105
106            # Make a GetFeature request to the WFS service
107            response = wfs.getfeature(typename=type_name,
108                                      outputFormat=output_format,
109                                      startindex=start_index,
110                                      sortby=sort_by,
111                                      maxfeatures=DATA_LOAD_FEATURES_PER_PROCESS)
112
113            data = response.read()
114            if 'json' in output_format.lower():
115                # Load the JSON features from the response
116                json_features = json.loads(data)
117
118                # Sometimes the server returns an empty feature set
119                if len(json_features['features']) == 0:
120                    # Log the successful loading of features
121                    logging.info(
122                        f"Loaded from {start_index} To {start_index + DATA_LOAD_FEATURES_PER_PROCESS}: {base_url}: {type_name}")
123                    # Return after successful loading
124                    return
125
126                # Create a GeoDataFrame from the JSON features with the specified CRS
127                crs = pyproj.CRS.from_epsg(int(epsg_code))
128                gdf = geopandas.GeoDataFrame.from_features(json_features, crs=crs)
129
130                try:
131                    # Construct the PostgreSQL connection URL
132                    postgres_url = f"postgresql://{config('db_user')}:{config('db_password')}@{config('db_host')}:{config('db_port')}/{config('db_name')}"
133
134                    # Create the SQLAlchemy engine
135                    engine = create_engine(postgres_url, poolclass=NullPool)
136
137                    # Use the connection for GeoDataFrame.to_postgis
138                    gdf.to_postgis(name=table_name, con=engine, schema='public', if_exists='append')
139
140                    # Explicitly close the engine
141                    engine.dispose()
142                except Exception as e:
143                    # Handle any errors that occur during saving to PostGIS
144                    logging.error(f"Error saving to PostGIS: {e}")
145                    error_event.set()
146                    DataLoader.set_loading_error(self_url,
147                                                 f"Failed saving to PostGIS from {start_index} To {start_index + DATA_LOAD_FEATURES_PER_PROCESS}: {base_url}: {type_name}: {e}")
148                    return
149            elif 'gml' in output_format.lower():
150                # gdf = geopandas.read_file(StringIO(data.decode('utf-8')), driver='GML')
151                save_gml_to_db(data, table_name, 'append')
152
153            # Log the successful loading of features
154            logging.info(
155                f"Loaded from {start_index} To {start_index + DATA_LOAD_FEATURES_PER_PROCESS}: {base_url}: {type_name}")
156
157            # Return after successful loading
158            return
159        except Exception as e:
160            # traceback.print_exc()
161            # Log the retry attempt in case of an error
162            logging.info(
163                f"Try loading again from {start_index} To {start_index + DATA_LOAD_FEATURES_PER_PROCESS}: {base_url}: {type_name}: {tries}: {e}")
164            tries += 1
165
166    # If all retries fail, set the error event and raise an exception
167    error_event.set()
168    DataLoader.set_loading_error(self_url,
169                                 f"Failed loading from {start_index} To {start_index + DATA_LOAD_FEATURES_PER_PROCESS}: {base_url}: {type_name}")
170    logging.info(
171        f"Failed loading from {start_index} To {start_index + DATA_LOAD_FEATURES_PER_PROCESS}: {base_url}: {type_name}")

186        """
187        Check if this data loader is able to process the data at the URL.
188
189        Parameters:
190        - url (str): The URL to be validated.
191
192        Returns:
193        - bool: True if the loader can process the data, False otherwise.
194        """
195        try:
196            # Parse the url and check the query parameter typename exists and is valid
197            parsed_url = urlparse(url)
198            query_params = parse_qs(parsed_url.query)
199            query_params = {key.lower(): value for key, value in query_params.items()}
200            if 'typename' in query_params.keys():
201                typename = query_params['typename'][0]
202                wfs = WebFeatureService(url, "1.1.0")  # we will support if only 1.1.0 is implemented
203                layers = wfs.contents.keys()
204
205                # Check if the layer with the typename exists
206                # MapServer WFS may not have a name space with a layer
207                return typename in layers or typename.split(':')[-1] in layers
208        except requests.exceptions.HTTPError as e:
209            return False

213    def normalize_url(self):
214        parsed = urlparse(self.url)
215        query = sorted([(key.lower(), value) for key, value in parse_qsl(parsed.query)])
216        # this WFS loader only needs a typename; other parameters will be generated based on the capabilities.
217        query = [(key, value) for key, value in query if key == 'typename']
218        parsed = ParseResult(parsed.scheme,
219                             parsed.netloc,
220                             parsed.path,
221                             parsed.params,
222                             urlencode(query),
223                             parsed.fragment)
224        return parsed.geturl()

227    def __get_sort_by(fields):
228        id_fields = [key for key, value in fields.items() if
229                     key.lower().endswith('id') and (value == 'long' or value == 'int' or value == 'double')]
230        if id_fields:
231            return id_fields[0]
232
233        string_id_fields = [key for key, value in fields.items() if key.lower().endswith('id') and value == 'string']
234        if string_id_fields:
235            return string_id_fields[0]
236
237        return list(fields.keys())[0]

248        hits_xml = fromstring(response.content)

261        if 'numberReturned' in hits_xml.attrib:
262            try:
263                return int(fromstring(response.content).attrib['numberReturned'])
264            except Exception:
265                pass

271        try:

273            capabilities = str(response.read())

285                return 'ArcGIS'

287            return 'Unknown'

289        except Exception as e:

291            return 'Unknown'

293    def load(self):
294        """
295        Load data served by WFS into the database and update the data status.
296
297        This method loads data into the specified table and then updates the status
298        of the data associated with the URL to 'Saved' in the mediator's data status table.
299        """
300        # Get base url
301        parsed_url = urlparse(self.url)
302        base_url = f"{parsed_url.scheme}://{parsed_url.netloc}{parsed_url.path}"
303
304        # Get all query parameters
305        query_params = parse_qs(parsed_url.query)
306        query_params = {key.lower(): value for key, value in query_params.items()}
307
308        # Check if the query parameter typename occurs
309        if 'typename' in query_params.keys():
310            typename = query_params['typename'][0]
311        else:
312            raise DataLoaderError(f'Missing the parameter typename')
313
314        # Get the version of this WFS
315        wfs = WebFeatureService(base_url, '1.1.0', timeout=120)
316        version = wfs.identification.version
317        # logging.info(wfs.identification.__dict__)
318
319        # Get the server vendor
320        vendor = self.__detect_server_vendor(wfs)
321        logging.info(f"Vendor: {vendor}")
322
323        # Get all layers
324        layers = wfs.contents.keys()
325        logging.info(layers)
326
327        # Check if the layer with the typename exists
328        # MapServer WFS may not have a name space in the layers
329        if typename not in layers and typename.split(':')[-1] not in layers:
330            raise DataLoaderError(f'{typename} not found')
331
332        # Get the schema of the typename
333        sort_by = None
334        if vendor == 'GeoServer' or vendor == 'Unknown':
335            schema = wfs.get_schema(typename)
336            logging.info(f"schema: {schema['properties']}")
337            sort_by = self.__get_sort_by(schema['properties'])
338            logging.info(f"sort_by: {sort_by}")
339
340        # Starting index for all the features
341        start_index = 0
342
343        # check the allowed output formats for GetFeature and choose json or gml if exists
344        output_format = 'application/json'
345        get_feature = wfs.getOperationByName("GetFeature")
346        output_formats = get_feature.parameters['outputFormat']['values']
347        logging.info(f"allowed output formats: {output_formats}")
348        json_formats = [fmt for fmt in output_formats if 'json' in fmt.lower()]
349        json_formats.sort(key=len)
350        if json_formats:
351            output_format = json_formats[0]
352        else:
353            gml_formats = [fmt for fmt in output_formats if 'gml' in fmt.lower()]
354            if gml_formats:
355                output_format = gml_formats[0]
356        logging.info(f"output format: {output_format}")
357
358        # Special setting for ArcGIS
359        if vendor == 'ArcGIS':
360            # for ArcGIS WFS, GetFeature supports geojson when using the version 2.0.0,
361            # even it is not specified in GetCapabilities
362            wfs = WebFeatureService(base_url, '2.0.0', timeout=60)
363            version = '2.0.0'
364            output_format = 'geojson'
365            logging.info(f"change the output format to {output_format} for ArcGIS")
366
367        # Get the total feature number
368        # Note: found several cases in which the actual feature number is smaller than
369        # the declared total feature number
370        total = self.__get_total_feature_count(base_url, typename, version)
371
372        # Get the projection. Sometimes returned features may not associate with an epsg code.
373        epsg_code = wfs.contents[typename].crsOptions[0].code
374        crs = pyproj.CRS.from_epsg(int(epsg_code))
375
376        # Fetch the initial features
377        response = wfs.getfeature(typename=typename,
378                                  outputFormat=output_format,
379                                  startindex=start_index,
380                                  sortby=sort_by,
381                                  maxfeatures=DATA_LOAD_INIT_FEATURES)
382        data = response.read()
383        if 'json' in output_format.lower():
384            # Create a GeoDataFrame from the JSON features with the specified CRS
385            data_json = json.loads(data)
386            gdf = geopandas.GeoDataFrame.from_features(data_json, crs=crs)
387
388            # Construct the PostgresSQL connection URL
389            postgres_url = f"postgresql://{config('db_user')}:{config('db_password')}@{config('db_host')}:{config('db_port')}/{config('db_name')}"
390
391            # Create the SQLAlchemy engine.
392            # Note: using NullPool to prevent SQLAlchemy starting a connection pool.
393            engine = create_engine(postgres_url, poolclass=NullPool)
394
395            # Use the connection for GeoDataFrame.to_postgis.
396            # Note: use 'replace' strategy to initialize the table
397            gdf.to_postgis(name=self.table_name, con=engine, schema='public', if_exists='replace')
398
399            # Explicitly close the engine
400            engine.dispose()
401
402        elif 'gml' in output_format.lower():
403            # GeoPandas is very buggy to load GML; Use ogr2ogr instead
404            # gdf = geopandas.read_file(StringIO(data.decode('utf-8')), driver='GML', crs=crs)
405            save_gml_to_db(data, self.table_name, 'overwrite')
406
407        logging.info(f"Loaded from {start_index} To {start_index + DATA_LOAD_INIT_FEATURES} out of {total}: {self.url}")
408
409        # Create an event to signal an error by processes
410        error_event = Event()
411
412        # Start more processes to save the next WFS_LOAD_FEATURES_PER_PROCESS features
413        start_index = start_index + DATA_LOAD_INIT_FEATURES;
414        processes = []
415        while total > start_index and not error_event.is_set():
416            process = Process(target=process_load_features,
417                              args=(self.url,
418                                    base_url,
419                                    version,
420                                    typename,
421                                    epsg_code,
422                                    start_index,
423                                    sort_by,
424                                    self.table_name,
425                                    output_format,
426                                    vendor,
427                                    error_event))
428            processes.append(process)
429            process.start()
430            logging.info(
431                f"Start loading {start_index} to {start_index + DATA_LOAD_FEATURES_PER_PROCESS} out of {total}: {self.url}")
432            start_index = start_index + DATA_LOAD_FEATURES_PER_PROCESS
433
434            # To be nice to the remote server and the mediator
435            # Start limited new processes to load data
436            if len(processes) == DATA_LOAD_MAX_PROCESSES:
437                for process in processes:
438                    process.join()
439
440                # Clear the processes list
441                processes.clear()
442
443        # Complete the left processes
444        if len(processes) > 0 and not error_event.is_set():
445            for process in processes:
446                process.join()
447
448        # Check if an error occurred in any process
449        if error_event.is_set():
450            logging.info(f"An error occurred in one of the processes: {self.url}")
451            return
452
453        logging.info(f"Completed data loading: {self.url}")
454
455        # Update the status
456        DataLoader.update_data_status(self.url, 'Saved')

Args: self_url (str): The original URL for error logging base_url (str): The base URL of the WFS service. version (str): The version of the WFS service (e.g., '1.1.0', '2.0.0'). type_name (str): The name of the feature type to retrieve from the WFS. epsg_code (str): The EPSG code representing the coordinate reference system (CRS). start_index (int): The starting index for retrieving features in a paginated manner. sort_by (str): The attribute to use for sorting features. table_name (str): The name of the PostgresSQL table to store the features. output_format (str): The name for the JSON output format. vendor (str): The name of the server vendor. error_event (threading.Event): An event to signal if an error occurs during the data loading process.
